// ---- CONFIGURACOES -----------------------------------------
var CFG = {
  emailRemetente: 'Portal Previne',
  whatsapp: {
    ativo:  false,
    apiKey: 'SUA_APIKEY_AQUI'
  }
};

// Mapeamento dc-d01..dc-d31 => nome legivel (cabecalho da planilha)
var DOC_LABELS = {
  'dc-d01': 'Ficha de Cadastro',
  'dc-d02': 'Declaracao de Dependentes (IR)',
  'dc-d03': 'Termo de Opcao Vale Transporte',
  'dc-d04': 'Carteira de Identidade (RG)',

var DOC_IDS = ['dc-d01','dc-d02','dc-d03','dc-d04','dc-d05','dc-d06','dc-d07',
               'dc-d08','dc-d09','dc-d10','dc-d11','dc-d12','dc-d13','dc-d14',
               'dc-d15','dc-d16','dc-d17','dc-d18','dc-d19','dc-d20','dc-d21',
               'dc-d22','dc-d23','dc-d24','dc-d25','dc-d26','dc-d27','dc-d28',

// ------------------------------------------------------------

function doGet(e) {
  var cb     = e.parameter.callback || 'cb';
  var action = e.parameter.action   || '';
  try {
    var ss  = SpreadsheetApp.getActiveSpreadsheet();
    var now = Utilities.formatDate(new Date(), 'America/Sao_Paulo', 'dd/MM/yyyy HH:mm:ss');

    // --- SALVAR CREDENCIAIS DO USUARIO (CPF + email + senha) ---
    if (action === 'addUsuario') {
      var su2 = sheet(ss, 'Usuarios', ['CPF','Email','Senha','Data'], '#374151');
      var cpfNorm = normCPF(p(e,'cpf'));
      // Update if exists, else append
      var urows = su2.getDataRange().getValues();
      var ufound = false;
      for (var ui = 1; ui < urows.length; ui++) {
        if (normCPF(urows[ui][0]) === cpfNorm) {
          su2.getRange(ui+1, 1).setValue(cpfNorm);
          su2.getRange(ui+1, 2).setValue(p(e,'email'));
          su2.getRange(ui+1, 3).setValue(p(e,'senha'));
          ufound = true; break;
        }
      }
      if (!ufound) {
        su2.appendRow([cpfNorm, p(e,'email'), p(e,'senha'), now]);
      }
    }

    // --- BUSCAR CREDENCIAIS DE UM USUARIO ---
    if (action === 'getUsuario') {
      var su3 = ss.getSheetByName('Usuarios');
      if (su3) {
        var urows3 = su3.getDataRange().getValues();
        var cpfGet = normCPF(p(e,'cpf'));
        for (var ui3 = 1; ui3 < urows3.length; ui3++) {
          if (normCPF(urows3[ui3][0]) === cpfGet) {
            // Also get profile from Candidatos
            var perfil = {};
            var sc3 = ss.getSheetByName('Candidatos');
            if (sc3) {
              var prows = sc3.getDataRange().getValues();
              for (var pi = 1; pi < prows.length; pi++) {
                if (normCPF(prows[pi][1]) === cpfGet) {
                  perfil = {
                    nome: String(prows[pi][3]||''), tel: String(prows[pi][4]||''),
                    rua:  String(prows[pi][5]||''), bairro: String(prows[pi][6]||''),
                    cidade: String(prows[pi][7]||''), estado: String(prows[pi][8]||''),
                    cep:  String(prows[pi][9]||''), funcao: String(prows[pi][10]||''),
                    curriculo: String(prows[pi][11]||''), cpf: String(prows[pi][1]||'')
                  };
                  break;
                }
              }
            }
            // Get docs from Documentos sheet
            var docsU = {};
            var sdU = ss.getSheetByName('Documentos');
            if (sdU && sdU.getLastRow() > 1) {
              var drowsU = sdU.getDataRange().getValues();
              for (var dui = 1; dui < drowsU.length; dui++) {
                if (normCPF(drowsU[dui][1]) === cpfGet) {
                  for (var dci2 = 4; dci2 < drowsU[0].length && dci2 < drowsU[dui].length; dci2++) {
                    var fn2 = String(drowsU[dui][dci2]||'');
                    var si2 = dci2 - 4;
                    if (fn2 && fn2 !== 'Nao enviado' && si2 < DOC_IDS.length) {
                      docsU[DOC_IDS[si2]] = { name: fn2 };
                    }
                  }
                  break;
                }
              }
            }
            var senhaStored = String(urows3[ui3][2]||'').trim();
            return jsonp(cb, {
              status: 'ok',
              found: true,
              email: String(urows3[ui3][1]||'').trim(),
              senha: senhaStored,
              perfil: perfil,
              docs: docsU
            });
          }
        }
      }
      return jsonp(cb, { status: 'ok', found: false });
    }

    // --- CADASTRO DO CANDIDATO ---
    if (action === 'addCandidato') {
      var sc = sheet(ss, 'Candidatos', [
        'Data/Hora','CPF','E-mail','Nome','Telefone',
        'Endereco','Bairro','Cidade','Estado','CEP','Funcao','Curriculo'
      ], '#1565C0');
      sc.appendRow([now, normCPF(p(e,'cpf')), p(e,'email'), p(e,'nome'), p(e,'tel'),
        p(e,'rua'), p(e,'bairro'), p(e,'cidade'), p(e,'estado'),
        p(e,'cep'), p(e,'funcao'), p(e,'curriculo')]);

      var nomeCad   = p(e,'nome');
      var funcaoCad = p(e,'funcao');
      var emailCad  = p(e,'email');
      var telCad    = p(e,'tel');

      var corpoCad = 'Ola, ' + nomeCad + '!\n\n'
        + 'Seu cadastro foi recebido com sucesso pelo Portal de Vagas Previne.\n\n'
        + 'Funcao desejada: ' + funcaoCad + '\n'
        + 'E-mail: '          + emailCad  + '\n'
        + 'Telefone: '        + telCad    + '\n\n'
        + 'Em breve nossa equipe de RH analisara seu perfil e entrara em contato.\n\n'
        + '\n--\n'
        + 'Este e-mail e enviado automaticamente. Por favor, nao responda.\n'
        + 'Em caso de duvidas, entre em contato diretamente com o RH da Previne.\n\n'
        + 'Atenciosamente,\nEquipe Previne\nEngenharia & Gestao de Riscos';
      enviarEmail(emailCad, 'Cadastro recebido - Portal Previne', corpoCad);

      if (CFG.whatsapp.ativo) {
        enviarWhatsApp(limparTel(telCad),
          'Portal Previne\n\nOla, ' + nomeCad + '! Seu cadastro foi recebido.\n'
          + 'Funcao: ' + funcaoCad + '\nEm breve o RH entrara em contato!');
      }
    }

    // --- DOCUMENTOS ENVIADOS ---
    if (action === 'addDocumentos') {
      // Build headers: fixed cols + label of each doc slot
      var hdrs = ['Data/Hora','CPF','Nome','Funcao'];
      for (var hi = 0; hi < DOC_IDS.length; hi++) {
        hdrs.push(DOC_LABELS[DOC_IDS[hi]] || DOC_IDS[hi]);
      }
      var sd = sheet(ss, 'Documentos', hdrs, '#7C3AED');

      // Build row: for each slot, use filename or 'Nao enviado'
      var rowData = [now, p(e,'cpf'), p(e,'nome'), p(e,'funcao')];
      var enviados = [];
      for (var di = 0; di < DOC_IDS.length; di++) {
        var fname = p(e, DOC_IDS[di]);
        rowData.push(fname !== '' ? fname : 'Nao enviado');
        if (fname !== '') enviados.push(DOC_LABELS[DOC_IDS[di]] + ': ' + fname);
      }
      sd.appendRow(rowData);

      // Busca email e tel na aba Candidatos
      var emailDoc = p(e,'email') || '';
      var telDoc   = '';
      var scDoc = ss.getSheetByName('Candidatos');
      if (scDoc) {
        var rowsDoc = scDoc.getDataRange().getValues();
        for (var i = 1; i < rowsDoc.length; i++) {
          if (String(rowsDoc[i][1]) === p(e,'cpf')) {
            if (!emailDoc) emailDoc = rowsDoc[i][2];
            telDoc = rowsDoc[i][4];
            break;
          }
        }
      }

      var listaEnv = enviados.length > 0 ? enviados.join('\n') : 'Nenhum arquivo identificado';

      if (emailDoc) {
        var corpoDoc = 'Ola, ' + p(e,'nome') + '!\n\n'
          + 'Seus documentos foram recebidos com sucesso:\n\n'
          + listaEnv + '\n\n'
          + 'Nossa equipe ira analisar cada documento e voce sera notificado sobre o status.\n\n'
          + '\n--\n'
          + 'Este e-mail e enviado automaticamente. Por favor, nao responda.\n'
          + 'Em caso de duvidas, entre em contato diretamente com o RH da Previne.\n\n'
          + 'Atenciosamente,\nEquipe Previne\nEngenharia & Gestao de Riscos';
        enviarEmail(emailDoc, 'Documentos recebidos - Portal Previne', corpoDoc);
      }

      if (CFG.whatsapp.ativo && telDoc) {
        enviarWhatsApp(limparTel(telDoc),
          'Portal Previne\n\nOla, ' + p(e,'nome') + '! Seus documentos foram recebidos.\n'
          + enviados.length + ' arquivo(s) registrado(s).\n\nAguarde a analise do RH.');
      }
    }

    // --- ENVIAR CODIGO POR EMAIL ---
    if (action === 'enviarCodigoPorEmail') {
      var emailDest = p(e,'email');
      var nomeDest  = p(e,'nome');
      var codDest   = p(e,'codigo');
      if (emailDest && codDest) {
        var corpoCode = 'Ola, ' + nomeDest + '!\n\n'
          + 'O RH gerou um codigo de liberacao para voce acessar a area de documentos.\n\n'
          + 'Seu codigo: ' + codDest + '\n\n'
          + 'Acesse o portal, va ate a aba Documentos e insira este codigo para desbloquear o envio.\n\n'
          + '\n--\n'
          + 'Este e-mail e enviado automaticamente. Por favor, nao responda.\n'
          + 'Em caso de duvidas, entre em contato diretamente com o RH da Previne.\n\n'
          + 'Atenciosamente,\nEquipe Previne\nEngenharia & Gestao de Riscos';
        enviarEmail(emailDest, 'Seu codigo de acesso - Portal Previne', corpoCode);
      }
    }

    // --- CODIGO RH GERADO ---
    if (action === 'addCodigo') {
      var s3 = sheet(ss, 'Codigos', [
        'Data/Hora','Candidato','CPF','Codigo','Status'
      ], '#F59E0B');
      s3.appendRow([now, p(e,'nome'), p(e,'cpf'), p(e,'codigo'), 'Aguardando uso']);
    }

    // --- APROVACAO / RECUSA DE DOCUMENTO ---
    if (action === 'updateDocStatus') {
      var su = sheet(ss, 'Aprovacoes', [
        'Data/Hora','CPF','Nome','Documento','Arquivo','Status','Analista'
      ], '#DC2626');
      su.appendRow([now, p(e,'cpf'), p(e,'nome'), p(e,'documento'),
        p(e,'arquivo'), p(e,'status'), 'RH']);

      var emailAp = p(e,'email') || '';
      if (!emailAp) {
        var scAp = ss.getSheetByName('Candidatos');
        if (scAp) {
          var rowsAp = scAp.getDataRange().getValues();
          for (var ia = 1; ia < rowsAp.length; ia++) {
            if (String(rowsAp[ia][1]) === p(e,'cpf')) { emailAp = rowsAp[ia][2]; break; }
          }
        }
      }

      if (emailAp) {
        var stLabel   = p(e,'status');
        var isAprov   = (stLabel === 'Aprovado');
        var prefixo   = isAprov ? '[APROVADO]' : '[RECUSADO]';
        var resultMsg = isAprov
          ? 'Parabens! Este documento foi aprovado pelo RH.'
          : 'Este documento foi recusado. Acesse o portal e reenvie o arquivo correto.';
        var corpoAp = 'Ola, ' + p(e,'nome') + '!\n\n'
          + 'O RH analisou um dos seus documentos:\n\n'
          + 'Documento : ' + p(e,'documento') + '\n'
          + 'Arquivo   : ' + p(e,'arquivo')   + '\n'
          + 'Status    : ' + stLabel          + '\n\n'
          + resultMsg + '\n\n'
          + 'Em caso de duvidas, entre em contato com o setor de RH.\n\n'
          + '\n--\n'
          + 'Este e-mail e enviado automaticamente. Por favor, nao responda.\n'
          + 'Em caso de duvidas, entre em contato diretamente com o RH da Previne.\n\n'
          + 'Atenciosamente,\nEquipe Previne\nEngenharia & Gestao de Riscos';
        enviarEmail(emailAp, prefixo + ' Documento - Portal Previne', corpoAp);

        if (CFG.whatsapp.ativo) {
          var telAp = '';
          var scAp2 = ss.getSheetByName('Candidatos');
          if (scAp2) {
            var rowsAp2 = scAp2.getDataRange().getValues();
            for (var ib = 1; ib < rowsAp2.length; ib++) {
              if (String(rowsAp2[ib][1]) === p(e,'cpf')) { telAp = rowsAp2[ib][4]; break; }
            }
          }
          if (telAp) enviarWhatsApp(limparTel(telAp),
            'Portal Previne\n\nOla, ' + p(e,'nome') + '!\n'
            + p(e,'documento') + ': ' + stLabel + '\n' + resultMsg);
        }
      }
    }

    // --- LER CANDIDATOS DA PLANILHA ---
    if (action === 'getCandidatos') {
      var result = [];
      var sc2 = ss.getSheetByName('Candidatos');
      if (sc2 && sc2.getLastRow() > 1) {
        var rows = sc2.getDataRange().getValues();
        // cols: 0=Data, 1=CPF, 2=Email, 3=Nome, 4=Tel, 5=Rua, 6=Bairro, 7=Cidade, 8=Estado, 9=CEP, 10=Funcao, 11=Curriculo
        for (var ri = 1; ri < rows.length; ri++) {
          var r = rows[ri];
          if (!r[1]) continue; // skip empty rows
          result.push({
            cpf:      normCPF(r[1]),
            email:    String(r[2] || ''),
            nome:     String(r[3] || ''),
            tel:      String(r[4] || ''),
            rua:      String(r[5] || ''),
            bairro:   String(r[6] || ''),
            cidade:   String(r[7] || ''),
            estado:   String(r[8] || ''),
            cep:      String(r[9] || ''),
            funcao:   String(r[10] || ''),
            curriculo:String(r[11] || '')
          });
        }
      }
      // Also get docs counts from Documentos sheet
      var sdocs = ss.getSheetByName('Documentos');
      var docsMap = {};
      if (sdocs && sdocs.getLastRow() > 1) {
        var drows = sdocs.getDataRange().getValues();
        var dhdrs = drows[0]; // header row
        for (var dri = 1; dri < drows.length; dri++) {
          var dr = drows[dri];
          var dcpf = String(dr[1] || '');
          if (!dcpf) continue;
          if (!docsMap[dcpf]) docsMap[dcpf] = {};
          // cols 4+ are doc slots (dc-d01..dc-d31)
          for (var dci = 4; dci < dhdrs.length && dci < dr.length; dci++) {
            var fname = String(dr[dci] || '');
            if (fname && fname !== 'Nao enviado' && fname !== '') {
              // find the slot id from DOC_IDS index
              var slotIdx = dci - 4;
              if (slotIdx < DOC_IDS.length) {
                docsMap[dcpf][DOC_IDS[slotIdx]] = { name: fname };
              }
            }
          }
        }
      }
      // Get approval status from Aprovacoes sheet
      var saprov = ss.getSheetByName('Aprovacoes');
      var aprovMap = {};
      if (saprov && saprov.getLastRow() > 1) {
        var aprovRows = saprov.getDataRange().getValues();
        for (var ari = 1; ari < aprovRows.length; ari++) {
          var ar = aprovRows[ari];
          var acpf = String(ar[1] || '');
          var adoc  = String(ar[4] || ''); // arquivo (filename)
          var astat = String(ar[5] || ''); // status
          if (!acpf || !adoc) continue;
          if (!aprovMap[acpf]) aprovMap[acpf] = {};
          // find slot id by filename
          for (var asi = 0; asi < DOC_IDS.length; asi++) {
            if (docsMap[acpf] && docsMap[acpf][DOC_IDS[asi]] && docsMap[acpf][DOC_IDS[asi]].name === adoc) {
              aprovMap[acpf][DOC_IDS[asi]] = astat === 'Aprovado' ? 'approved' : 'rejected';
            }
          }
        }
      }
      // Get codes from Codigos sheet
      var scod = ss.getSheetByName('Codigos');
      var codesResult = [];
      if (scod && scod.getLastRow() > 1) {
        var crows = scod.getDataRange().getValues();
        for (var cri = 1; cri < crows.length; cri++) {
          var cr = crows[cri];
          codesResult.push({
            createdAt: String(cr[0] || ''),
            nome:      String(cr[1] || ''),
            cpf:       String(cr[2] || ''),
            codigo:    String(cr[3] || ''),
            usedAt:    String(cr[4] || '')
          });
        }
      }
      return jsonp(cb, { status: 'ok', candidatos: result, docs: docsMap, aprovacoes: aprovMap, codigos: codesResult });
    }

    // --- BUSCAR CODIGO POR VALOR ---
    if (action === 'getCode') {
      var scod2 = ss.getSheetByName('Codigos');
      if (scod2 && scod2.getLastRow() > 1) {
        var cvals = scod2.getDataRange().getValues();
        for (var ci2 = 1; ci2 < cvals.length; ci2++) {
          if (String(cvals[ci2][3]) === p(e,'codigo')) {
            return jsonp(cb, {
              status:    'ok',
              found:     true,
              cpf:       normCPF(cvals[ci2][2]),
              nome:      String(cvals[ci2][1]||''),
              codigo:    String(cvals[ci2][3]||''),
              createdAt: String(cvals[ci2][0]||''),
              usedAt:    String(cvals[ci2][4]||'')
            });
          }
        }
      }
      return jsonp(cb, { status: 'ok', found: false });
    }

    // --- MARCAR CODIGO COMO USADO ---
    if (action === 'usarCodigo') {
      var s4 = ss.getSheetByName('Codigos');
      if (s4) {
        var vals = s4.getDataRange().getValues();
        for (var iv = 1; iv < vals.length; iv++) {
          if (vals[iv][3] === p(e,'codigo')) {
            s4.getRange(iv+1, 5).setValue('Utilizado em ' + now);
            break;
          }
        }
      }
    }

    return jsonp(cb, { status: 'ok', action: action });
  } catch(err) {
    return jsonp(cb, { status: 'error', msg: err.toString() });
  }
}

// --- ENVIO DE E-MAIL ---
function enviarEmail(destino, assunto, corpo) {
  try {
    if (!destino || destino.indexOf('@') === -1) return;
    GmailApp.sendEmail(destino, assunto, corpo, { name: CFG.emailRemetente });
  } catch(err) {
    Logger.log('Erro email: ' + err.toString());
  }
}

// --- ENVIO WHATSAPP via CallMeBot ---
function enviarWhatsApp(numero, mensagem) {
  try {
    if (!numero || !CFG.whatsapp.ativo || CFG.whatsapp.apiKey === 'SUA_APIKEY_AQUI') return;
    var url = 'https://api.callmebot.com/whatsapp.php'
      + '?phone='  + numero
      + '&text='   + encodeURIComponent(mensagem)
      + '&apikey=' + CFG.whatsapp.apiKey;
    UrlFetchApp.fetch(url);
  } catch(err) {
    Logger.log('Erro WhatsApp: ' + err.toString());
  }
}

// --- HELPERS ---
function limparTel(tel) {
  var d = tel.replace(/[^0-9]/g, '');
  if (!d) return '';
  if (d.length === 11) return '55' + d;
  if (d.length === 10) return '55' + d;
  return d;
}

function p(e, k) {
  var v = e.parameter[k];
  return (v === undefined || v === 'undefined' || v === null) ? '' : String(v);
}

// Normalize CPF: strip non-digits, pad to 11 chars with leading zeros
function normCPF(val) {
  var s = String(val || '').replace(/[^0-9]/g, '');
  while (s.length < 11) s = '0' + s;
  return s;
}

function jsonp(cb, obj) {
  return ContentService
    .createTextOutput(cb + '(' + JSON.stringify(obj) + ')')
    .setMimeType(ContentService.MimeType.JAVASCRIPT);
}

// Creates sheet if not exists; always ensures header row is correct
function sheet(ss, name, headers, color) {
  var s = ss.getSheetByName(name);
  if (!s) {
    s = ss.insertSheet(name);
    s.appendRow(headers);
    var hdrRange = s.getRange(1, 1, 1, headers.length);
    hdrRange.setFontWeight('bold').setBackground(color).setFontColor('#FFFFFF');
    hdrRange.setWrap(true);
    s.setFrozenRows(1);
    s.setRowHeight(1, 40);
    try { s.autoResizeColumns(1, headers.length); } catch(e) {}
  } else {
    // Ensure header row exists and is styled (in case sheet was created before)
    var existing = s.getRange(1, 1, 1, s.getLastColumn()).getValues()[0];
    if (!existing[0] || existing[0] === '') {
      s.insertRowBefore(1);
      s.getRange(1, 1, 1, headers.length).setValues([headers])
       .setFontWeight('bold').setBackground(color).setFontColor('#FFFFFF');
      s.setFrozenRows(1);
    }
  }
  return s;
}
